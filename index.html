// CẤU HÌNH ĐƯỜNG DẪN ĐẾN SHEET CỦA BẠN const GOOGLE_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbxA7PfNhCsvUBCvejeAFZvEbEt-vSqrPargEHeI5yN9YLy1Lu9aKyxSxmVO-FhktzjX2g/exec"; const CURRENT_USER = "User1"; let html5QrCode; let isScanning = false; let currentQRText = ""; let scanHistory = []; let cameraTrack = null; // Cấu hình quét cơ bản, an toàn cho mọi thiết bị const scanConfig = { fps: 15, qrbox: { width: 250, height: 250 } }; function startScanner() { // Khởi tạo đối tượng quét trên khung "reader" html5QrCode = new Html5QrCode("reader"); // Dùng cấu hình facingMode cơ bản giống bản code đầu tiên để chống treo máy html5QrCode.start( { facingMode: "environment" }, scanConfig, onScanSuccess ) .then(() => { isScanning = true; document.getElementById('status').innerText = "Sẵn sàng quét..."; document.getElementById('status').style.color = "green"; // Sau khi camera mở thành công, kiểm tra xem có hỗ trợ Zoom không setTimeout(setupZoomFeature, 1000); // Trì hoãn 1 giây để phần cứng ổn định }) .catch(err => { document.getElementById('status').innerText = "Không thể mở camera. Hãy kiểm tra quyền truy cập!"; document.getElementById('status').style.color = "red"; console.error(err); }); } function setupZoomFeature() { try { // Tìm luồng video đang chạy trong thẻ reader const videoElement = document.querySelector("#reader video"); if (!videoElement) return; const stream = videoElement.srcObject; if (!stream) return; cameraTrack = stream.getVideoTracks()[0]; const capabilities = cameraTrack.getCapabilities(); const zoomSlider = document.getElementById("zoom-slider"); // Nếu điện thoại và trình duyệt hỗ trợ tính năng Zoom if (capabilities && "zoom" in capabilities) { zoomSlider.min = capabilities.zoom.min; zoomSlider.max = capabilities.zoom.max; zoomSlider.step = capabilities.zoom.step; zoomSlider.value = capabilities.zoom.min; zoomSlider.disabled = false; // Lắng nghe sự kiện kéo thanh gạt zoom zoomSlider.oninput = function(e) { if (cameraTrack) { cameraTrack.applyConstraints({ advanced: [{ zoom: parseFloat(e.target.value) }] }).catch(err => console.log("Lỗi áp dụng zoom:", err)); } }; } else { document.getElementById("zoom-container").innerHTML = "Thiết bị không hỗ trợ gạt Zoom trực tiếp"; } } catch (e) { console.log("Không thiết lập được tính năng zoom:", e); document.getElementById("zoom-container").innerHTML = "Không hỗ trợ gạt Zoom trên thiết bị này"; } } function onScanSuccess(decodedText, decodedResult) { if (!isScanning) return; isScanning = false; currentQRText = decodedText; // Hiển thị giao diện 3 nút, ẩn camera document.getElementById('camera-section').style.display = 'none'; document.getElementById('control-panel').style.display = 'block'; document.getElementById('qr-info').innerText = decodedText; document.getElementById('status').innerText = "Vui lòng chọn hành động..."; document.getElementById('status').style.color = "orange"; } function sendData(actionType) { document.getElementById('status').innerText = `Đang gửi lệnh ${actionType}...`; document.getElementById('status').style.color = "orange"; const vietnamTime = new Date().toLocaleString("vi-VN", { timeZone: "Asia/Ho_Chi_Minh" }); fetch(GOOGLE_SCRIPT_URL, { method: "POST", mode: "no-cors", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ qrContent: currentQRText, timestamp: vietnamTime, action: actionType, user: CURRENT_USER }) }) .then(() => { document.getElementById('status').innerText = `Đã lưu thành công: ${actionType}!`; document.getElementById('status').style.color = "green"; updateHistory(vietnamTime, currentQRText, actionType); resumeScanner(); }) .catch(err => { document.getElementById('status').innerText = "Lỗi kết nối. Vui lòng bấm lại!"; document.getElementById('status').style.color = "red"; isScanning = false; }); } function resumeScanner() { document.getElementById('control-panel').style.display = 'none'; document.getElementById('camera-section').style.display = 'block'; document.getElementById('status').innerText = "Sẵn sàng quét..."; document.getElementById('status').style.color = "green"; // Trả thanh gạt zoom về x1 khi quét mã mới const zoomSlider = document.getElementById("zoom-slider"); if (zoomSlider && !zoomSlider.disabled && cameraTrack) { zoomSlider.value = zoomSlider.min; cameraTrack.applyConstraints({ advanced: [{ zoom: parseFloat(zoomSlider.min) }] }).catch(e => {}); } isScanning = true; } function updateHistory(time, content, action) { scanHistory.unshift({ time, content, action }); if (scanHistory.length > 10) scanHistory.pop(); const listContainer = document.getElementById('history-list'); listContainer.innerHTML = ""; scanHistory.forEach(item => { const itemDiv = document.createElement('div'); itemDiv.className = "history-item"; const shortContent = item.content.length > 22 ? item.content.substring(0, 22) + "..." : item.content; const badgeClass = item.action === 'Add' ? 'badge-add' : 'badge-delete'; const timeOnly = item.time.split(" ")[1] || ""; itemDiv.innerHTML = ` [${timeOnly}] ${shortContent} ${item.action} `; listContainer.appendChild(itemDiv); }); } // Tự động kích hoạt khi tải xong trang window.addEventListener("DOMContentLoaded", startScanner);
